\subsection*{(3)}
\subsubsection*{解}
\begin{align*}
  D(E)=\frac{L^3(2m)^{\frac{3}{2}}}{2\pi^2\hbar^3}\sqrt{E}
\end{align*}
\hrulefill
\subsubsection*{解説}
与式から
\begin{align}
  D(E){\rm d}E&=\Omega(E+{\rm d}E)-\Omega(E)\\
  D(E)&=\frac{\Omega(E+{\rm d}E)-\Omega(E)}{{\rm d}E}\\
  &=\frac{{\rm d}}{{\rm d} E}\Omega(E)\\
  &=\frac{L^3(2m)^{\frac{3}{2}}}{2\pi^2\hbar^3}\sqrt{E}
\end{align}
